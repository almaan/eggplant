14def ztest( 

15 x1: np.ndarray, 

16 x2: np.ndarray, 

17 se1: float, 

18 se2: float, 

19 delta: float = 0, 

20 alpha: float = 0.05, 

21 two_sided: bool = True, 

22) -> Dict[str, Union[float, bool]]: 

23 

24 n1 = x1.shape[0] 

25 n2 = x2.shape[0] 

26 div = np.sqrt(se1 / n1 + se2 / n2) 

27 top = x1.mean() - x2.mean() - delta 

28 

29 z = np.divide(top, div) 

30 pval = 2 * (1 - norm(0, 1).cdf(np.abs(z))) 

31 is_sig = pval < alpha 

32 

33 return dict(z=z, pvalue=pval, sig=is_sig)

36def mixed_normal( 

37 mus: np.ndarray, 

38 vrs: np.ndarray, 

39 ws: Optional[np.ndarray] = None, 

40) -> Tuple[np.ndarray, np.ndarray]: 

41 """mean and var for weighted mixed 

42 normal. 

43 

44 For n distributions :math:`N_i(\mu_i,\sigma^2_i)` 

45 we compute the mean and variance for the new weighted 

46 mix: 

47 

48 :math:`N(\mu_{new},\sigma^2_{new}) = \sum_{i=1}^n w_iN(\mu_i,\sigma^2_i)` 

49 

50 :param mus: mean values for each sample 

51 :type mus: np.ndarray 

52 :param vrs: variance values for each sample 

53 :type vrs: np.ndarray 

54 :param ws: weights to use when computing the 

55 new mean for the mixed distribution. 

56 :type ws: Optional[np.ndarray] 

57 

58 :return: a tuple being :math:`(\mu_{new},\sigma^2_{new})` 

59 :rtype: Tuple[np.ndarray,np.ndarray] 

60 

61 

62 """ 

63 

64 N = mus.shape[1] 

65 

66 if ws is None: 

67 ws = np.ones(N) / N 

68 

69 if len(mus.shape) == 2: 

70 ws = ws.reshape(1, N) 

71 

72 v1 = np.sum(ws * vrs, axis=1) 

73 v2 = np.sum(ws * (mus ** 2), axis=1) 

74 v3 = np.sum(ws * mus, axis=1) ** 2 

75 

76 new_var = v1 + v2 - v3 

77 

78 new_mean = np.sum(ws * mus, axis=1) 

79 

80 return (new_mean, new_var)

83def sdea( 

84 data: Union[ad.AnnData, "m.Reference"], 

85 group_col: str, 

86 n_std: int = 2, 

87 subset: Optional[Dict[str, Any]] = None, 

88 weights: Optional[np.ndarray] = None, 

89) -> Dict[str, Dict[str, np.ndarray]]: 

90 """spatial differential expression analysis 

91 

92 conduct spatial differential expression analysis 

93 (sDEA) 

94 

95 :param data: object (either anndata or reference) containing 

96 the spatial profiles to be compared. 

97 :type data: Union[ad.AnnData, "m.Reference"] 

98 :param group_col: column to make comparison with respect to 

99 :type group_col: str 

100 :param n_std: number of standard deviations that should be used when testing 

101 for differential expression. If the interval mean_1 +/- n_std*std_1 

102 overlaps with the interval mean_2 +/- n_std*std_2 the features are 

103 considered as non-differentially expressed, defaults to 2 

104 :type n_std: int 

105 :param subset: subset groups in the contrastive analysis, for example 

106 `subset={feature:value}` will only compare those profiles where the value 

107 of *feature* is *value*, defaults to no subsampling 

108 :type subset: Optional[Dict[str, Any]] 

109 :param weights: n_samples vector of weights, where the i:th value of the 

110 vector indicates the weight that should be assigned to each sample in the 

111 sdea analysis, default to 1/n_samples. 

112 :type weights: Optional[np.ndarray] 

113 :return: a dictionary where each analyzed feature is an entry, and each 

114 entry is a dictionary with two values: `diff` being the spot-wise difference 

115 between the samples, and `sig` being an indicator of whether the difference 

116 is significant or not. 

117 :rtype: Dict[str, Dict[str, np.ndarray]] 

118 

119 """ 

120 

121 if isinstance(data, ad.AnnData): 

122 adata = data 

123 elif isinstance(data, m.Reference): 

124 adata = data.adata 

125 else: 

126 raise TypeError("The provided data type is not yet supported.") 

127 

128 if subset is not None: 

129 keep_idx = set(list(range(adata.shape[0]))) 

130 for col, val in subset.items(): 

131 if col in adata.var.columns.values: 

132 sidx = set(np.where(adata.var[col].values == val)[0]) 

133 keep_idx = keep_idx.intersection(sidx) 

134 else: 

135 raise ValueError(f"{col} is not found in the meta data") 

136 

137 keep_idx = list(keep_idx) 

138 

139 if keep_idx: 

140 adata = adata[:, keep_idx] 

141 

142 group_labels = adata.var[group_col] 

143 group_labels = ut.pd_to_np(group_labels) 

144 uni_group_labels = np.unique(group_labels) 

145 n_labels = len(uni_group_labels) 

146 

147 res = OrderedDict() 

148 

149 def _get_mixed(label): 

150 idx_x = group_labels == uni_group_labels[label] 

151 mus_x = adata.X[:, idx_x] 

152 vars_x = adata.layers["var"][:, idx_x] 

153 wvals_x = mixed_normal(mus_x, vars_x, ws=weights) 

154 return wvals_x 

155 

156 for k1 in range(n_labels - 1): 

157 wval1 = _get_mixed(k1) 

158 for k2 in range(k1 + 1, n_labels): 

159 wval2 = _get_mixed(k2) 

160 

161 wmus = np.hstack((wval1[0][:, np.newaxis], wval2[0][:, np.newaxis])) 

162 wvars = np.hstack((wval1[1][:, np.newaxis], wval2[1][:, np.newaxis])) 

163 ordr = np.argsort(wmus, axis=1) 

164 wmus = np.take_along_axis(wmus, ordr, axis=1) 

165 wvars = np.take_along_axis(wvars, ordr, axis=1) 

166 wstds = wvars ** 0.5 

167 

168 comp = wmus[:, 0] + n_std * wstds[:, 0] < wmus[:, 1] - n_std * wstds[:, 1] 

169 

170 name = uni_group_labels[k1] + "_vs_" + uni_group_labels[k2] 

171 

172 diff = wval1[0] - wval2[0] 

173 res[name] = dict(diff=diff, sig=comp) 

174 

175 return res
