578 """Constructor class

581 :type cmap: Dict[T,str]

583 """ 

584 

585 self.n_c = len(cmap) 

586 self.cdict = cmap 

587 if not all([isinstance(x, int) for x in self.cdict.keys()]): 

588 self.numeric_cdict = {k: c for k, c in enumerate(self.cdict.keys())} 

589 else: 

590 self.numeric_cdict = self.cdict 

591 

592 def __call__( 

593 self, 

594 x: Union[T, np.ndarray], 

595 n_elements: bool = False, 

596 ) -> Union[str, np.ndarray]: 

597 

598 if n_elements: 

599 n = x 

600 clr = [self.numeric_cdict[ii % self.n_c] for ii in range(n)] 

601 

602 elif hasattr(x, "__len__") and not isinstance(x,str): 

603 if isinstance(x, (tuple, list)): 

604 uni = set(x) 

605 has_keys = all([k in self.cdict.keys() for k in uni]) 

606 if has_keys: 

607 clr = [self.cdict[ii] for ii in x] 

608 else: 

609 raise ValueError("Keys not supported") 

610 else: 

611 n = len(x) 

612 clr = [self.numeric_cdict[ii % self.n_c] for ii in range(n)] 

613 clr = np.array(clr) 

614 else: 

615 if x in self.cdict.keys(): 

616 clr = self.cdict[x] 

617 else: 

618 raise ValueError(f"{x} is not supported as value.") 

619

786) -> Tuple[plt.Figure, plt.Axes]:

910 bbox_style: Optional[Dict[str, Any]] = None, 

911 return_figures: bool = False, 

912) -> Optional[Tuple[plt.Figure, plt.Axes]]: 

913 """function to visualize landmark distribution 

914 with given set of parameters. Here Poisson Disc 

915 Sampling (PSD) is used to randomly distribute the landmarks 

916 within the spatial domain. 

917 

918 :param adata: data to visualize

920 :param feature: feature to visualize,

927 :py:paramref:`~eggplant.methods.estimate_n_landmarks.params.spread_distance` 

928 :type spread_distance: Optional[float] 

929 :param diameter_multiplier: see 

930 :py:paramref:`~eggplant.methods.estimate_n_landmarks.diameter_multiplier` 

931 :type diameter_multiplier: float 

932 :param side_size: side size of plot, defaults to 5 

933 :type side_size: float

949 marker, if None then `spread_distance / 5`, defaults to None 

950 :type text_h_offset: Optional[float]

955

962 

963 if spread_distance is None: 

964 diameter = ut.get_capture_location_diameter(adata) 

965 if diameter: 

966 spread_distance = diameter * diameter_multiplier
