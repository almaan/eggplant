502class PoissonDiscSampler: 

503 """Poisson Disc Sampler 

504 Designed according to the principles 

505 outlined in :cite:t:`pds` but for d=2. 

506 """ 

507 

508 def __init__( 

509 self, 

510 crd: np.ndarray, 

511 min_dist: float, 

512 seed: Optional[int] = None, 

513 ) -> None: 

514 """Constructor method 

515 

516 :param crd: coordinates of the data, a bounding box 

517 (rectangular) will be created from these coordinates. 

518 Samples will be contained within this bounding box. 

519 :type crd: np.ndarray 

520 :param min_dist: minimal allowed distance 

521 between two samples, no two samples will be 

522 closer than this value. 

523 :type min_dist: float 

524 :param seed: random seed (numpy) 

525 :type seed: int 

526 

527 

528 """ 

529 

530 self.seed = seed 

531 

532 self.r = min_dist 

533 self.r2 = self.r ** 2 

534 self.delta = self.r / np.sqrt(2) 

535 

536 self._set_grid(crd) 

537 self._reset() 

538 

539 def _set_grid( 

540 self, 

541 _crd: np.ndarray, 

542 ) -> None: 

543 

544 """build grid to guide neighbor search""" 

545 crd = _crd.copy() 

546 self.x_correct = _crd[:, 0].min() 

547 self.y_correct = _crd[:, 1].min() 

548 crd[:, 0] -= self.x_correct 

549 crd[:, 1] -= self.y_correct 

550 self.x_max = crd[:, 0].max() 

551 self.y_max = crd[:, 1].max() 

552 self.center = crd.mean(axis=0) 

553 

554 self.n_x = int(crd[:, 0].max() / self.delta) 

555 self.n_y = int(crd[:, 1].max() / self.delta) 

556 

557 def _reset( 

558 self, 

559 ) -> None: 

560 """reset sampler""" 

561 self.cells = {x: -1 for x in product(range(self.n_x + 1), range(self.n_y + 1))} 

562 self.active = [0] 

563 self.samples = [self.center] 

564 pos = self.coord_to_cell(self.center) 

565 self.cells[pos] = 0

567 def coord_to_cell( 

568 self, 

569 point: Union[np.ndarray, Tuple[float, float]], 

570 ) -> Tuple[int, int]: 

571 """helper function, transforms coordinates 

572 to cell id (in grid) 

573 

574 :param point: coordinates of point to get cell id for 

575 :type point: Union[np.ndarray, Tuple[float, float]] 

576 

577 :return: a tuple (i,j) where i is the grid row index 

578 and j is the column index. 

579 :rtype: Tuple[int,int] 

580 

581 """ 

582 _x = int(point[0] // self.delta) 

583 _y = int(point[1] // self.delta) 

584 return (_x, _y)

586 def add_k_in_annulus( 

587 self, 

588 point: Union[np.ndarray, Tuple[float, float]], 

589 k: int = 5, 

590 ) -> np.ndarray: 

591 """adds k points randomly in annulus around point 

592 

593 :param point: point to create annulus around 

594 :type point: Union[np.ndarray, Tuple[float, float]], 

595 :param k: number of points to add to annulus, default 5 

596 :type k: int 

597 

598 :return: array of coordinates with added points 

599 :rtype: np.ndarray 

600 

601 """ 

602 xy = [] 

603 while len(xy) < k: 

604 _r = np.random.uniform(self.r, 2 * self.r) 

605 _theta = np.random.uniform(0, 2 * np.pi) 

606 _x = _r * np.cos(_theta) 

607 _y = _r * np.sin(_theta) 

608 tmp = point + (_x, _y) 

609 if (tmp[0] > 0 and tmp[0] < self.x_max) and ( 

610 tmp[1] > 0 and tmp[1] < self.y_max 

611 ): 

612 

613 xy.append(tmp) 

614 

615 return np.array(xy)

617 def get_neighbors( 

618 self, 

619 idx: Tuple[int, int], 

620 ) -> List[Tuple[int, int]]: 

621 """get neighbors in grid 

622 

623 Note: includes self. 

624 

625 :param idx: index of cell to get neighbors of 

626 :type idx: Tuple[int,int] 

627 

628 :return: list of neighbor indices 

629 :rtype: List[Tuple[int,int]] 

630 

631 """ 

632 res = [] 

633 for px in range(-2, 3): 

634 for py in range(-2, 3): 

635 new_x = idx[0] + px 

636 new_y = idx[1] + py 

637 if (new_x >= 0 and new_x <= self.n_x) and ( 

638 new_y >= 0 and new_y <= self.n_y 

639 ): 

640 res.append((new_x, new_y)) 

641 

642 return res

644 def sample( 

645 self, 

646 max_points: Optional[int] = None, 

647 k: Optional[int] = 4, 

648 ) -> np.ndarray: 

649 """sample from domain 

650 

651 :param max_points: max number of points 

652 to sample from domain. 

653 :type max_points: Optional[int] 

654 :param k: number of points to position 

655 in annulus, default 4 

656 :type k: Optional[int] 

657 

658 :return: array of samples from domain 

659 :rtype: np.ndarray 

660 

661 """ 

662 

663 if self.seed is not None: 

664 np.random.seed(self.seed) 

665 

666 if max_points is None: 

667 max_points = np.inf 

668 elif max_points > len(self.cells): 

669 print("[WARNING] : You specified more points than the domain allows") 

670 

671 while len(self.samples) < max_points and len(self.active) > 0: 

672 

673 idx = np.random.choice(self.active) 

674 point = self.samples[idx] 

675 new_points = self.add_k_in_annulus(point, k=k) 

676 inactivate = True 

677 for new_point in new_points: 

678 

679 new_point_cell = self.coord_to_cell(new_point)
